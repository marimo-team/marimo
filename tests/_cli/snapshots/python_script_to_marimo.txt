"""A simple Python script."""

import marimo


app = marimo.App()


@app.cell
def _():
    import sys

    def main():
        print("Hello, World!")
    return main, sys


@app.cell
def _(main, sys):
    def _main_():
        main()
        sys.exit(0)

    _main_()
    return


if __name__ == "__main__":
    app.run()
