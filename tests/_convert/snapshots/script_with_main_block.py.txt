"""Script with main block."""

import marimo

__generated_with = "0.0.0"
app = marimo.App()


@app.cell
def _():
    import sys
    import argparse

    def parse_args():
        parser = argparse.ArgumentParser()
        parser.add_argument("name", help="Name to greet")
        return parser.parse_args()

    def greet(name):
        print(f"Hello, {name}!")
    return greet, parse_args, sys


@app.cell
def _(greet, parse_args, sys):
    def _main_():
        args = parse_args()
        greet(args.name)
        sys.exit(0)

    _main_()
    return


if __name__ == "__main__":
    app.run()
