    r"""
    return
    """,
    name="_"
)

    r"""
    # Single quote string
    x = "hello \"world\"

    # Triple quote string
    x = "hello \"\"\"world\"\"\""

    # Triple quote string with other slashes
    x = "hello \"\"\"world\"\"\"

    """,
    name="_"
)


if __name__ == "__main__":
    app.run()
