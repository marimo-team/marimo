# col<angles
_base_chart = (
    alt.Chart(df)
    .transform_aggregate(count="count()", groupby=["col<angles"])
    .transform_window(
        rank="rank()",
        sort=[
            alt.SortField("count", order="descending"),
            alt.SortField("col<angles", order="ascending"),
        ],
    )
    .transform_filter(alt.datum.rank <= 10)
    .transform_joinaggregate(total_count="sum(count)")
    .transform_calculate(
        percentage="datum.count / datum.total_count"
    )
    .encode(
        y=alt.Y(
            "col<angles:N",
            sort="-x",
            axis=alt.Axis(title=None),
        ),
        x=alt.X("count:Q"),
        tooltip=[
            alt.Tooltip("col<angles:N"),
            alt.Tooltip("count:Q", format=",.0f"),
        ],
    )
)

_bar_chart = _base_chart.mark_bar()
_text_chart = _base_chart.mark_text(align="left", dx=3).encode(
    text=alt.Text("percentage:Q", format=".2%")
)

_chart = (
    (_bar_chart + _text_chart)
    .properties(width="container")
    .configure_view(stroke=None)
    .configure_axis(grid=False)
)
_chart
